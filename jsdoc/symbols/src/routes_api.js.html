312 function modifyPassword(req) {
313   dn = 'cn='+req.body.username+' ,ou='+req.body.team+', o=ldap';
314   change = new ldap.Change({
315     operation: 'replace', modification: { userpassword: req.body.username.userPassword }
316   });
317   clientModify(dn, change);
318 }

320 function modifyState(dn, toChange) {
321   change = new ldap.Change({
322     operation: 'replace', modification: { state: toChange }
323   });
324   clientModify(dn, change);

327 function modifyToken(dn, toChange) {
